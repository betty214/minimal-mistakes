[12100번 : 2048(EASY)](https://www.acmicpc.net/problem/12100)

- 코드

```java
package backjun;

import java.util.Scanner;

public class ex_12100 {
	public static int [][]arr;
	public static int n;
	public static int []news=new int[4];
	public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        n = sc.nextInt();
        arr = new int[n][n];
        for(int i=0;i<n;i++) {
        	for(int j=0;j<n;j++) {
        		arr[i][j]=sc.nextInt();
        	}
        }
        for(int i=0;i<4;i++) {
			news[i]=0;
		}
        where();
        for(int i=0;i<4;i++) {
			System.out.print(news[i]+" ");
		}
        for(int i=0;i<n;i++) {
        	for(int j=0;j<n;j++) {
        		System.out.print(arr[i][j]);
        	}System.out.println();
        }
        System.out.println(findMax());
	}
	public static void up() {
		for(int i=0;i<n-1;i++) {
			for(int j=0;j<n;j++) {
				if(arr[i][j]==arr[i+1][j]) {
					arr[i][j]*=2;
					arr[i+1][j]=0;
				}
			}
		}
		for(int i=0;i<n-1;i++) {
			for(int j=0;j<n;j++) {
				if(arr[i][j]==0) {
					arr[i][j]=arr[i+1][j];
					arr[i+1][j]=0;
				}
			}
		}
		
	}
	public static void left() {
		for(int i=0;i<n;i++) {
			for(int j=0;j<n-1;j++) {
				if(arr[i][j]==arr[i][j+1]) {
					arr[i][j]*=2;
					arr[i][j+1]=0;
				}
			}
		}
		for(int i=0;i<n;i++) {
			for(int j=0;j<n-1;j++) {
				if(arr[i][j]==0) {
					arr[i][j]=arr[i][j+1];
					arr[i][j+1]=0;
				}
			}
		}
	}
	public static void down() {
		for(int i=n-1;i>0;i--) {
			for(int j=0;j<n;j++) {
				if(arr[i][j]==arr[i-1][j]) {
					arr[i][j]*=2;
					arr[i-1][j]=0;
				}
			}
		}
		for(int i=n-1;i>0;i--) {
			for(int j=0;j<n;j++) {
				if(arr[i][j]==0) {
					arr[i][j]=arr[i-1][j];
					arr[i-1][j]=0;
				}
			}
		}
		
	}
	public static void right() {
		for(int i=0;i<n;i++) {
			for(int j=n-1;j>0;j--) {
				if(arr[i][j]==arr[i][j-1]) {
					arr[i][j]*=2;
					arr[i][j-1]=0;
				}
			}
		}
		for(int i=0;i<n;i++) {
			for(int j=n-1;j>0;j--) {
				if(arr[i][j]==0) {
					arr[i][j]=arr[i][j-1];
					arr[i][j-1]=0;
				}
			}
		}
	}
	public static int findMax() {
		int max=0;
		for(int i=0;i<n;i++) {
        	for(int j=0;j<n;j++) {
        		if (max<arr[i][j]) {
        			max=arr[i][j];
        		}
        	}
        }
		return max;
	}
	public static void where() {
		
		for(int i=0;i<n-1;i++) {
			for(int j=0;j<n;j++) {
				if(arr[i][j]==arr[i+1][j]) {
					news[0]++;
				}
			}
		}
		for(int i=n-1;i>0;i--) {
			for(int j=0;j<n;j++) {
				if(arr[i][j]==arr[i-1][j]) {
					news[1]++;
				}
			}
		}
		for(int i=0;i<n;i++) {
			for(int j=n-1;j>0;j--) {
				if(arr[i][j]==arr[i][j-1]) {
					news[2]++;
				}
			}
		}
		for(int i=0;i<n;i++) {
			for(int j=n-1;j>0;j--) {
				if(arr[i][j]==arr[i][j-1]) {
					news[3]++;
				}
			}
		}
	}
}
```
