- Z

[1074번 : Z](https://www.acmicpc.net/problem/1074)

- 코드

```java
package backjun;

import java.util.Scanner;

public class ex_1074 {
	public static int N,r,c,n,index;
	public static int [][] arr;
    public static void main(String[] args) {
    	Scanner sc = new Scanner(System.in);
		N = sc.nextInt();
		r = sc.nextInt();
		c = sc.nextInt();
		int n=1,index=0;
		for(int i=0;i<N;i++) {
			n*=2;
		}
		arr=new int[n][n];
		for(int i=0;i<n;i++) {
			for(int j=0;j<n;j++) {
				arr[i][j]=0;
			}
		}
		DFS(n,0,0);
    }
    public static void DFS(int k, int si, int sj) {
        if (k==1) {
        	if(si==r&sj==c) {
            	System.out.println(index);
        	}
        	index++;
        	return;
		} else {
			int m=k/2;
			for (int i = si; i < si + k; i += m) {
				for (int j = sj; j < sj + k; j += m) {
					DFS(m, i, j);
				}
			}
		}
	}
}
```
